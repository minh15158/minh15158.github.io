Mar 19, 2021
1 min read 
# usr/local/Homebrew
cd "$(brew --repo)"	
git remote set-url origin git://mirrors.ustc.edu.cn/brew.git

# usr/local/Homebrew/Library/Taps/homebrew/homebrew-core
cd "$(brew --repo)/Library/Taps/homebrew/homebrew-core"
git remote set-url origin git://mirrors.ustc.edu.cn/homebrew-core.git

# usr/local/Homebrew/Library/Taps/homebrew/homebrew-cask
cd "$(brew --repo)/Library/Taps/homebrew/homebrew-cask"
git remote set-url origin git://mirrors.ustc.edu.cn/homebrew-cask.git

# usr/local/Homebrew
git -C "$(brew --repo)" remote set-url origin https://mirrors.ustc.edu.cn/brew.git
git -C "$(brew --repo)" remote -v

# usr/local/Homebrew/Library/Taps/homebrew/homebrew-core
git -C "$(brew --repo homebrew/core)" remote set-url origin https://mirrors.ustc.edu.cn/homebrew-core.git
git -C "$(brew --repo homebrew/core)" remote -v

# usr/local/Homebrew/Library/Taps/homebrew/homebrew-cask
git -C "$(brew --repo homebrew/cask)" remote set-url origin https://mirrors.ustc.edu.cn/homebrew-cask.git
git -C "$(brew --repo homebrew/cask)" remote -v

# /usr/local/Homebrew/Library/Taps/homebrew/homebrew-cask-fonts
# https://github.com/Homebrew/homebrew-cask-fonts
git -C "$(brew --repo homebrew/cask-fonts)" remote set-url origin
git -C "$(brew --repo homebrew/cask-fonts)" remote -v

# /usr/local/Homebrew/Library/Taps/homebrew/homebrew-cask-drivers
# https://github.com/Homebrew/homebrew-cask-drivers
git -C "$(brew --repo homebrew/cask-drivers)" remote set-url origin
git -C "$(brew --repo homebrew/cask-drivers)" remote -v

# /usr/local/Homebrew/Library/Taps/homebrew/homebrew-cask-versions
# https://github.com/Homebrew/homebrew-cask-versions
git -C "$(brew --repo homebrew/cask-versions)" remote set-url origin
git -C "$(brew --repo homebrew/cask-versions)" remote -v

echo 'export HOMEBREW_BOTTLE_DOMAIN=https://mirrors.ustc.edu.cn/homebrew-bottles/' >> ~/.zshrc
source ~/.zshrc

echo 'export HOMEBREW_BOTTLE_DOMAIN=https://mirrors.ustc.edu.cn/homebrew-bottles/' >> ~/.bash_profile
source ~/.bash_profile
